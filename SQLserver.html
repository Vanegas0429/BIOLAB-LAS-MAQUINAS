Ejemplos CRUD

-- Crear
CREATE TABLE productos (
  id INT IDENTITY(1,1) PRIMARY KEY,
  nombre NVARCHAR(100),
  precio DECIMAL(10,2)
);

-- Consultar
SELECT * FROM productos;

-- Actualizar
UPDATE productos SET precio=99.99 WHERE id=1;

-- Eliminar
DELETE FROM productos WHERE id=1;
