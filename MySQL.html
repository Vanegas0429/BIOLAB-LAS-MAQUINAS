Características

  • Base de datos relacional de código abierto.
  • Compatible con SQL estándar.
  • Muy usada en aplicaciones web.
  • Escalable y multiplataforma.

Ejemplos CRUD

-- Insertar registro        
INSERT INTO usuarios (id, nombre, correo)
VALUES (1, 'Juan Pérez', 'juan@example.com');

-- Crear tabla
CREATE TABLE usuarios (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(100),
  email VARCHAR(100)
);

-- Consultar
SELECT * FROM usuarios;

-- Actualizar
UPDATE usuarios SET nombre='Carlos' WHERE id=1;

-- Eliminar
DELETE FROM usuarios WHERE id=1;
