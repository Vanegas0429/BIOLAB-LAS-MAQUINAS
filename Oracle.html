  • Muy segura y confiable para aplicaciones críticas.

Ejemplos CRUD

-- Crear
CREATE TABLE clientes (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  nombre VARCHAR2(100),
  ciudad VARCHAR2(100)
);

-- Consultar
SELECT * FROM clientes;

-- Actualizar
UPDATE clientes SET ciudad='Bogotá' WHERE id=1;

-- Eliminar
DELETE FROM clientes WHERE id=1;
